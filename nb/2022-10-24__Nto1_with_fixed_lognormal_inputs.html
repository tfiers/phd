izh = @eqs begin
    
    dv/dt = (k*(v-vₗ)*(v-vₜ) - u - I_syn) / C
    du/dt = a*(b*(v-vᵣ) - u)

    I_syn = gₑ*(v-Eₑ) + gᵢ*(v-Eᵢ)

    dgₑ/dt = -gₑ / τ  # Represents sum over all exc synapses
    dgᵢ/dt = -gᵢ / τ

has_spiked(vars, params) = begin
    @unpack v = vars
    @unpack vₛ = params

    return (v ≥ vₛ)
end;

on_self_spike!(vars, params) = begin
    @unpack vᵣ, Δu = params
    
    vars.v = vᵣ
    vars.u += Δu
end;
