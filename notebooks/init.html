Print precision for (floating point) numbers.

%precision %.4G

def set_np_precision(digits=4):
    # This allows eg `array([1000, 1E+05, 0.111])`
    np.set_printoptions(formatter={
        'float_kind': lambda x: format(x, f'.{digits}G'),
    })

set_np_precision(4)
